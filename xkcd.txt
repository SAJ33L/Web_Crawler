function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.parentNode.removeChild( dom_node ); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[ i ].parentNode.removeChild( dom_nodes[ i ] );
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
					+ '//' + window.location.hostname
					+ '&postid=1020'
					+ '&is_singular=';
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';
			var style_zap = document.createElement( 'style' );
			style_zap.type = 'text/css';
			style_zap.appendChild( document.createTextNode( css_zap ) );
			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );
			iframe.src = 'https://r-login.wordpress.com/remote-login.php?wpcom_remote_login=request_access&origin=' + encodeURIComponent( data.origin ) + '&wpcomid=' + encodeURIComponent( data.wpcomid );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			var style = document.createElement( 'style' );
			style.type = 'text/css';
			style.id = 'wpcom_request_access_styles';
			style.appendChild( document.createTextNode( css ) );
			document.body.appendChild( style );

			document.body.appendChild( iframe );
		}

		if ( data.msg === 'DONE' ) {
			wpcom_remote_login_final_cleanup();
		}
	}
}, false );

	<div class="entry-content">

You can preorder it now on <a href="https://www.amazon.com/How-Absurd-Scientific-Real-World-Problems/dp/0525537090/">Amazon</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/2/trackingcode/PRH5522E62429">Barnes &amp; Noble</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/6/trackingcode/penguinrandom">IndieBound</a>, and <a href="https://itunes.apple.com/us/book/how-to/id1451461524?mt=11">Apple Books</a>.</em></p>



<p><em>Here&#8217;s an excerpt from the How To chapter on file transfers.</em></p>



<h3 class="wp-block-heading">Chapter 19: How to Send a File</h3>



<p>Sending large data files can be difficult.</p>



<p>Modern software systems have moved away from the concept of “files.” They don’t show you a folder full of image files; they show you a collection of photos. But files linger on, and will probably continue to do so for decades to come. And as long as we have files, we’ll need to send them to people.</p>



<div class="wp-block-image"><figure class="aligncenter size-large is-resized"><img data-attachment-id="977" data-permalink="https://blog.xkcd.com/sendafile_1/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png" data-orig-size="1673,469" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_1" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=640" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png" alt="" class="wp-image-977" width="598" height="167" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=596&amp;h=167 596w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=1191&amp;h=334 1191w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=150&amp;h=42 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=300&amp;h=84 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=768&amp;h=215 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_1.png?w=1024&amp;h=287 1024w" sizes="(max-width: 598px) 100vw, 598px" /></figure></div>



<p>The simplest, most obvious way to send a file is to pick up the device the file is stored on, walk over to the intended recipient, and hand it to them.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="978" data-permalink="https://blog.xkcd.com/sendafile_2/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png" data-orig-size="1482,302" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_2" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png" alt="" class="wp-image-978" width="540" height="109" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=535&amp;h=109 535w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=1070&amp;h=218 1070w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=150&amp;h=31 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=300&amp;h=61 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=768&amp;h=157 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_2.png?w=1024&amp;h=209 1024w" sizes="(max-width: 540px) 100vw, 540px" /></figure></div>



<p>Carrying computers can be difficult — especially the earlier ones that were the size of a whole room — so rather than carry the whole computer, you can try detaching a piece of the computer containing the file. You can then bring this piece to the other person and let them transfer it to their own device. On a desktop-style computer, the files may be stored on a hard drive, which can often be removed without destroying the computer.</p>



<p>On some devices, though, file storage is permanently attached to the electronics, making removal more challenging.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="980" data-permalink="https://blog.xkcd.com/sendafile_4/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png" data-orig-size="1346,397" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_4" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png" alt="" class="wp-image-980" width="596" height="176" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=596&amp;h=176 596w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=1192&amp;h=352 1192w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=150&amp;h=44 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=300&amp;h=88 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=768&amp;h=227 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_4.png?w=1024&amp;h=302 1024w" sizes="(max-width: 596px) 100vw, 596px" /></figure></div>



<p>A more convenient and less destructive solution is removable storage. You can make a copy of the file, put it on a device, then give the device to the person.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="981" data-permalink="https://blog.xkcd.com/sendafile_5/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png" data-orig-size="1301,666" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_5" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png" alt="" class="wp-image-981" width="484" height="248" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=484&amp;h=248 484w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=968&amp;h=496 968w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=150&amp;h=77 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=300&amp;h=154 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_5.png?w=768&amp;h=393 768w" sizes="(max-width: 484px) 100vw, 484px" /></figure></div>



<p>Carrying storage devices around is a surprisingly high-bandwidth way to transfer information. A suitcase full of MicroSD cards contains many petabytes of data; if you want to transfer very large amounts of data, mailing boxes of disk drives will <a href="https://what-if.xkcd.com/31/">almost always be faster</a> than transferring them over the internet.</p>



<p>If you want to send data to a specific location that’s too far to walk, but not convenient to reach by mail — say, a nearby mountaintop — you could try using some kind of autonomous vehicle to carry it. A delivery drone, for example, could easily carry a small satchel of SD cards containing terabytes of data.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="982" data-permalink="https://blog.xkcd.com/sendafile_6/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png" data-orig-size="1230,411" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_6" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png" alt="" class="wp-image-982" width="539" height="179" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=536&amp;h=179 536w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=1071&amp;h=358 1071w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=150&amp;h=50 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=300&amp;h=100 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=768&amp;h=257 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_6.png?w=1024&amp;h=342 1024w" sizes="(max-width: 539px) 100vw, 539px" /></figure></div>



<p>Quadcopter-style drones don’t work very well over long distances thanks to the limitations of batteries. If a drone has to carry its own battery, it can only hover for so long. If it wants to hover longer, it needs to carry a bigger battery, but that means more weight and faster power consumption. For the same reason that a house supported by jet engines <em>[Note: For more on hovering houses, see Chapter 7: How to Move]</em> can only hover for a few hours, small coaster-size drones typically have flight times measured in minutes, and the larger ones used for photography are usually limited to less than an hour in the air. Even if it flew very fast, a tiny drone carrying a MicroSD card could make it just a few miles before running out of steam.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="983" data-permalink="https://blog.xkcd.com/sendafile_7/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png" data-orig-size="773,1089" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_7" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=213" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png" alt="" class="wp-image-983" width="308" height="434" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=308&amp;h=434 308w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=616&amp;h=868 616w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=106&amp;h=150 106w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_7.png?w=213&amp;h=300 213w" sizes="(max-width: 308px) 100vw, 308px" /></figure></div>



<p>You could increase your range by making the drone bigger, adding solar panels, flying higher, and going faster. Or you could turn to the real masters of efficient long-distance flight:</p>



<p>Butterflies.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="989" data-permalink="https://blog.xkcd.com/sendafile_8/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png" data-orig-size="743,774" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_8" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png?w=288" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png" alt="" class="wp-image-989" width="290" height="301" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png?w=290&amp;h=301 290w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png?w=578&amp;h=602 578w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_8.png?w=144&amp;h=150 144w" sizes="(max-width: 290px) 100vw, 290px" /></figure></div>



<p>Monarch butterflies <a href="https://monarchjointventure.org/news-events/news/estimating-the-number-of-overwintering-monarchs-in-mexico">travel</a> thousands of miles during their migration across North America, with some traveling all the way from Canada to Mexico in a single season. If you look up during the spring or fall on the East Coast of the United States, you can sometimes spot them gliding by silently overhead, a few hundred feet above the ground. Their extreme range puts drones — and even many large aircraft — to shame.</p>



<p>You might think butterflies have an unfair advantage over battery-powered aerial vehicles, since they can stop to consume nectar and “recharge.” Butterflies will certainly refuel if they can, but they don’t necessarily need to. Another butterfly species, the painted lady (<em>Vanessa cardui</em>), is even more impressive: it flies <a href="https://academic.oup.com/biolinnean/article/120/2/274/2954916">from Europe to central Africa</a>, a <a href="https://royalsocietypublishing.org/doi/full/10.1098/rsbl.2016.0561">4,000-kilometer flight</a> that takes it over the Mediterranean Sea <em>and </em>the Sahara desert.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="984" data-permalink="https://blog.xkcd.com/sendafile_9/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png" data-orig-size="1017,882" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_9" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png" alt="" class="wp-image-984" width="474" height="411" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=474&amp;h=411 474w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=948&amp;h=822 948w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=150&amp;h=130 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=300&amp;h=260 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_9.png?w=768&amp;h=666 768w" sizes="(max-width: 474px) 100vw, 474px" /></figure></div>



<p>Butterflies make these journeys powered only by small reserves of stored lipids. They can fly so much more efficiently than drones in part by <a href="https://www.nrcresearchpress.com/doi/abs/10.1139/z79-180">soaring</a> — they seek out thermal columns and mountain waves, then hold their wings steady and ride the rising air upward like a vulture, hawk, or  eagle.</p>



<div class="wp-block-image"><figure class="aligncenter is-resized"><img data-attachment-id="986" data-permalink="https://blog.xkcd.com/sendafile_10/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png" data-orig-size="414,240" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="sendafile_10" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=414" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png" alt="" class="wp-image-986" width="169" height="98" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=169&amp;h=98 169w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=338&amp;h=196 338w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=150&amp;h=87 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/sendafile_10.png?w=300&amp;h=174 300w" sizes="(max-width: 169px) 100vw, 169px" /></figure></div>



<p>If you want to send your file to someone who lives along the migration route, could you get a butterfly to carry it for you?</p>



<p>Butterflies can carry weights. Volunteers with groups like <a href="https://www.monarchwatch.org">Monarch Watch</a> <a href="https://www.jstor.org/stable/3494598">tag</a> tens of thousands to hundreds of thousands of monarch butterflies each year to <a href="https://blog.nwf.org/2019/02/ how-the-monarch-butterfly-population-is-measured/">track their migration</a> and monitor their population (which has been in decline in recent decades). The smaller tags weigh about a milligram, but monarchs have completed their migration with larger tags that weigh 10 mg or more.</p>



<p>MicroSD cards weigh several hundred milligrams — comparable to the weight of a butterfly — so butterflies would have a hard time carrying them. But there’s no reason a storage device can’t be made smaller. MicroSD cards contain memory chips, and the storage density of these chips might be up to a gigabyte per square millimeter. Given those sizes, a butterfly could easily carry a tiny chip with a gigabyte of data. If your file is larger than that, you could break it up across multiple butterflies, and send multiple copies for redundancy.</p>



<p>When your data finally arrived at its destination, the recipient would have to check a lot of butterflies to assemble all the pieces of the file. You may need to develop some kind of touchless butterfly scanner that allows them to scan many butterflies at once.</p>



<p>You could avoid that problem — and increase your bandwidth dramatically — by using DNA-based storage. <a href="https://science.sciencemag.org/content/355/6328/950">Researchers</a> have stored data by encoding it into a DNA sample, then sequencing the DNA to recover it. Systems like this can achieve densities far beyond anything we do with chips — it’s possible to store and recover hundreds of petabytes of data using a single gram of DNA.</p>



<p>Each year, tens to hundreds of millions of monarch butterflies arrive in Mexico to spend the winter together in giant colonies in the mountains. If you tagged ten million of these butterflies with tiny pouches containing 5 mg of DNA storage each, the total capacity of the butterfly armada would be about 10 zettabytes — 10,000,000,000,000,000,000,000 bytes. That’s roughly the <a href="https://www.cisco.com/c/en/us/solutions/collateral/service-provider/global-cloud-index-gci/white-paper-c11-738085.html">total</a> amount of digital data in <em>existence </em>in the late 2010s.</p>



<p>If the Sun is warm, the winds are favorable, and it’s the right time of year, you could use butterflies to send someone the entire internet.</p>

<p><h2>Runners-up</h2></p>

	<div class="entry-content">
		
<p><em>My new book, <a href="https://xkcd.com/how-to/">How To</a>, comes out on September 3<sup>rd</sup> (preorder: <a href="https://www.amazon.com/How-Absurd-Scientific-Real-World-Problems/dp/0525537090/">Amazon</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/2/trackingcode/PRH5522E62429">Barnes &amp; Noble</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/6/trackingcode/penguinrandom">IndieBound</a>, and&nbsp;<a href="https://itunes.apple.com/us/book/how-to/id1451461524?mt=11">Apple Books</a>).</em></p>



<p><em>One of the most exciting things about writing </em>How To<em> was that, for a few chapters, I was able to reach out to some extremely cool people who were willing to apply their unique expertise to ridiculous tasks. Among those who generously agreed to help was <a href="https://www.serenaventures.com/">Serena Williams</a>. </em></p>



<p><em>Here&#8217;s a portion of the chapter “How to Catch a Drone”, in which Serena helped test whether tennis serves could be an effective countermeasure against flying robots &#8230; by taking a drone out onto a court and hitting tennis balls at it until it crashed.</em></p>



<hr class="wp-block-separator is-style-wide" />



<h4 class="wp-block-heading">How to Catch a Drone</h4>



<p>A wedding-photography drone is buzzing around above you. You don&#8217;t know what it&#8217;s doing there and you want it to stop.</p>



<div class="wp-block-image"><figure class="aligncenter size-large is-resized"><img data-attachment-id="931" data-permalink="https://blog.xkcd.com/drone_1/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png" data-orig-size="640,531" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="drone_1" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png" alt="" class="wp-image-931" width="309" height="256" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=309&amp;h=256 309w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=618&amp;h=512 618w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=150&amp;h=124 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_1.png?w=300&amp;h=249 300w" sizes="(max-width: 309px) 100vw, 309px" /></figure></div>



<p>Let’s suppose you have a garage full of sports equipment— baseballs, tennis rackets, lawn darts, you name it. Which sport’s projectiles would work best for hitting a drone? And who would make the best anti-drone guard? A baseball pitcher? A basketball player? A tennis player? A golfer? Someone else?</p>



<div class="wp-block-image"><figure class="aligncenter size-large is-resized"><img data-attachment-id="930" data-permalink="https://blog.xkcd.com/drone_4/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png" data-orig-size="1083,850" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="drone_4" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png" alt="" class="wp-image-930" width="619" height="485" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=619&amp;h=485 619w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=150&amp;h=118 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=300&amp;h=235 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=768&amp;h=603 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png?w=1024&amp;h=804 1024w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_4.png 1083w" sizes="(max-width: 619px) 100vw, 619px" /></figure></div>



<p>There are a few factors to consider — accuracy, weight, range, and projectile size.</p>



<div class="wp-block-image"><figure class="aligncenter size-large is-resized"><img data-attachment-id="929" data-permalink="https://blog.xkcd.com/drone_5/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png" data-orig-size="1278,702" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="drone_5" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=640" loading="lazy" src="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png" alt="" class="wp-image-929" width="719" height="394" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=717&amp;h=394 717w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=150&amp;h=82 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=300&amp;h=165 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=768&amp;h=422 768w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png?w=1024&amp;h=562 1024w, https://blog.xkcd.com/wp-content/uploads/2019/08/drone_5.png 1278w" sizes="(max-width: 719px) 100vw, 719px" /></figure></div>



<p>One sport I couldn’t find good data on was tennis. I found some studies of tennis pro accuracy, but they involved hitting targets marked on the court, rather than in the air.</p>



<p>So I reached out to Serena Williams.</p>



<p>To my pleasant surprise, she was happy to help out. Her husband, Alexis, offered a sacrificial drone, a DJI Mavic Pro 2 with a broken camera. They headed out to her practice court to see how effective the world’s best tennis player would be at fending off a robot invasion.</p>



<p>The few studies I could find suggested tennis players would score relatively low com- pared to athletes who threw projectiles— more like kickers than pitchers. My tentative guess was that a champion player would have an accuracy ratio around 50 when serving, and take 5–7 tries to hit a drone from 40 feet. (Would a tennis ball even knock down a drone? Maybe it would just ricochet off and cause the drone to wobble! I had so many questions.)</p>

<p>Alexis flew the drone over the net and hovered there, while Serena served from the baseline.</p>



<p>Her first serve went low. The second zipped past the drone to one side.</p>

<p>The third serve scored a direct hit on one of the propellers. The drone spun, momentarily seemed like it might stay in the air, then flipped over and smashed into the court. Serena started laughing as Alexis walked over to investigate the crash site, where the drone lay on the court near several propeller fragments.</p>

<h2>Introduction</h2>
<p>This is a book of bad ideas.</p>
<p>At least, most of them are bad ideas. It’s possible some good ones slipped through the cracks. If so, I apologize.</p>
<p>Some ideas that sound ridiculous turn out to be revolutionary. Smearing mold on an infected cut sounds like a terrible idea, but the discovery of penicillin showed that it could be a miracle cure. On the other hand, the world is full of disgusting stuff that you <em>could</em> smear on a wound, and most of them won’t make it better. Not all ridiculous ideas are good. So how do we tell the good ideas from the bad? We can try them and see what happens. But sometimes, we can use math, research, and things we already know to work out what will happen if we do.</p>
<p>When NASA was planning to send its car-size <em>Curiosity</em> rover to Mars, they had to figure out how to land it gently on the surface. Previous rovers had landed using parachutes and air bags, so NASA engineers considered this approach with <em>Curiosity</em>, but the rover was too large and heavy for parachutes to slow it down enough in Mars’s thin atmosphere. They also thought about mounting rockets on the rover to let it hover and touch down gently, but the exhaust would create dust clouds that would obscure the surface and make it hard to land safely.</p>
<p>Eventually, they came up with the idea of a <span style="font-weight:400;">“</span>sky crane<span style="font-weight:400;">”</span> —a vehicle that would hover high above the surface using rockets while lowering <em>Curiosity</em> to the ground on a long tether. This sounded like a ridiculous idea, but every other idea they could come up with was worse. The more they looked at the sky crane idea, the more plausible it seemed. So they tried it, and it worked.</p>
<p>We all start out life not knowing how to do things. If we’re lucky, when we need to do something, we can find someone to show us how. But sometimes, we have to come up with a way to do it ourselves. This means thinking of ideas and then trying to decide whether they’re good or not.</p>
<p>This book explores unusual approaches to common tasks, and looks at what would happen to you if you tried them. Figuring out why they would or wouldn’t work can be fun and informative and sometimes lead you to surprising places. Maybe an idea is bad, but figuring out exactly <em>why</em> it’s a bad idea can teach you a lot—and might help you think of a better approach.</p>
<p>And even if you already know the right way to do all these things, it can be helpful to try to look at the world through the eyes of someone who doesn’t. After all, for anything that <span style="font-weight:400;">“</span>everyone knows<span style="font-weight:400;">” </span>by the time they reach adulthood, every day over 10,000 people in the United States alone are learning it for the first time.</p>
<p><img data-attachment-id="911" data-permalink="https://blog.xkcd.com/2019/08/02/how-to-chapter-list-and-introduction/ten_thousand/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png" data-orig-size="778,440" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="ten_thousand" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png?w=300" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png?w=640" loading="lazy" class="alignnone size-full wp-image-911" src="https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png" alt="ten_thousand" width="778" height="440" srcset="https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png 778w, https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png?w=150&amp;h=85 150w, https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png?w=300&amp;h=170 300w, https://blog.xkcd.com/wp-content/uploads/2019/08/ten_thousand.png?w=768&amp;h=434 768w" sizes="(max-width: 778px) 100vw, 778px" /></p>
<p>That’s why I don’t like making fun of people for admitting they don’t know something or never learned how to do something. Because if you do that, all it does is teach them not to tell you when they’re learning something . . . and you miss out on the fun.</p>
<p>This book may not teach you how to throw a ball, how to ski, or how to move. But I hope you learn something from it. If you do, you’re one of today’s lucky 10,000.</p>

<h2><strong>How to invite me to your town</strong></h2>
<p>Don&#8217;t see your city on the list? You can invite me! I&#8217;ll be adding one more US stop to my tour based on the results of a challenge.</p>
<p>The challenge: <strong><em>Write the best story using nothing but book covers.</em></strong></p>
<p>Arrange the titles of your favorite books into sentences that tell a story, assemble a single continuous line of people holding up the covers, and take a photo or video documenting your feat. You can make the story as long as you want, but each book needs to be held by a different human.</p>

		<h2 class="entry-title"><a href="https://blog.xkcd.com/2019/02/05/how-to-absurd-scientific-advice-for-common-real-world-problems/" rel="bookmark">How To: Absurd Scientific Advice for Common Real-World&nbsp;Problems</a></h2>	</header><!-- .entry-header -->

	
	<div class="entry-content">
		<p><span style="font-weight:300;">I’m excited to announce that I have a new book coming out! It’s titled </span><b>How To: Absurd Scientific Advice for Common Real-World Problems</b><span style="font-weight:300;">, and it’s an extremely unhelpful guide to solving everyday problems.</span></p>
<p><a href="https://xkcd.com/how-to"><img data-attachment-id="884" data-permalink="https://blog.xkcd.com/2019/02/05/how-to-absurd-scientific-advice-for-common-real-world-problems/how-to-blog/" data-orig-file="https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png" data-orig-size="420,540" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="how-to-blog" data-image-description="" data-image-caption="" data-medium-file="https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png?w=233" data-large-file="https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png?w=420" loading="lazy" class="alignnone wp-image-884" src="https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png" alt="how-to-blog" width="300" height="386" srcset="https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png?w=300&amp;h=386 300w, https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png?w=117&amp;h=150 117w, https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png?w=233&amp;h=300 233w, https://blog.xkcd.com/wp-content/uploads/2019/02/how-to-blog.png 420w" sizes="(max-width: 300px) 100vw, 300px" /></a></p>
<p><span style="font-weight:300;"><strong>How To</strong> is an instruction manual for taking everyday problems and using science and creative thinking to turn them into much bigger and more exciting problems. It teaches you how to cross a river by boiling it, outlines some of the many uses for lava around the home, and walks you through how to use experimental military research to ensure that your friends will never again ask you to help them move. </span></p>
<p><span style="font-weight:300;">From changing a lightbulb to throwing a pool party, it describes unusual ways to accomplish common tasks, and analyzes what would happen to you if you tried them. In addition to being a profoundly unhelpful self-help book, it&#8217;s an exercise in applying math, science, and research to ordinary problems, and a tour through some of the strange and fun science underlying the world around us.</span></p>
<p><strong>How To</strong> will be released on September 3rd, but you can preorder it now from <a href="https://www.amazon.com/How-Absurd-Scientific-Real-World-Problems/dp/0525537090/">Amazon</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/2/trackingcode/PRH5522E62429">Barnes &amp; Noble</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/5/trackingcode/PRH5522E62429">Books-A-Million</a>, <a href="http://links.penguinrandomhouse.com/type/affiliate/isbn/9780525537090/siteID/8001/retailerid/6/trackingcode/penguinrandom">Indie Bound</a>, and <a href="https://itunes.apple.com/us/book/how-to/id1451461524?mt=11">Apple Books</a>.</p>
<p>For more, you can check out my <a href="https://ew.com/books/2019/02/05/randall-munroe-how-to-announcement/">Q&amp;A with Entertainment Weekly</a> about the book, and you can find details about foreign editions at <a href="https://xkcd.com/how-to/">xkcd.com/how-to</a>.</p>
